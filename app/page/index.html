Kyte API JS Documentation

Complete reference for the Kyte JavaScript SDK (v1.2.24). This documentation covers all available classes and methods for building Kyte applications.

Kyte Class - Main API Client
Constructor

Note: In published pages, the Kyte instance is automatically created as k (not kyte).

// This is done automatically in published pages:
var k = new Kyte(endpoint, public_key, identifier, account_number, app_identifier);
k.init();

Parameters:

  • endpoint - Backend API endpoint URL
  • public_key - API public key
  • identifier - API identifier
  • account_number - Account number
  • app_identifier - Application identifier
init()

Initialize Kyte instance by loading credentials from cookies. (Called automatically in published pages)

k.init();
sign(callback, error)

Generate HMAC signature for authenticated requests.

k.sign(
    (signature) => {
        console.log('Signature:', signature);
    },
    (error) => {
        console.error('Sign error:', error);
    }
);
CRUD Operations

get(model, field, value, headers, callback, error)

Retrieve records from the backend.

k.get('User', 'email', 'user@example.com', [],
    (response) => {
        console.log('User data:', response);
    },
    (error) => {
        console.error('Error:', error);
    }
);

post(model, data, formdata, headers, callback, error)

Create new records in the backend.

k.post('User',
    { name: 'John Doe', email: 'john@example.com' },
    null, [],
    (response) => {
        console.log('Created user:', response);
    },
    (error) => {
        console.error('Error:', error);
    }
);

put(model, field, value, data, formdata, headers, callback, error)

Update existing records in the backend.

k.put('User', 'id', '123',
    { name: 'Jane Doe' },
    null, [],
    (response) => {
        console.log('Updated user:', response);
    },
    (error) => {
        console.error('Error:', error);
    }
);

delete(model, field, value, callback, error)

Delete records from the backend (soft delete).

k.delete('User', 'id', '123',
    (response) => {
        console.log('Deleted user:', response);
    },
    (error) => {
        console.error('Error:', error);
    }
);
Session Management

sessionCreate(identity, callback, error, sessionController)

Create a new user session (login).

k.sessionCreate(
    { username: 'user@example.com', password: 'password123' },
    (response) => {
        console.log('Login successful:', response);
    },
    (error) => {
        console.error('Login failed:', error);
    }
);

sessionDestroy(error)

Destroy the current session (logout).

k.sessionDestroy((error) => {
    if (error) {
        console.error('Logout error:', error);
    } else {
        window.location.href = '/login.html';
    }
});

isSession(periodic, redir, interval)

Check if a valid session exists. Optionally enable periodic checking.

// Simple check
if (k.isSession(false, false)) {
    console.log('User is logged in');
}

// Periodic check with auto-redirect
k.isSession(true, true, 30000); // Check every 30 seconds

checkSession()

Check if session exists in cookies.

if (k.checkSession()) {
    console.log('Session cookie exists');
}
Utility Methods

setCookie(name, value, minutes, crossDomain)

Set a browser cookie.

k.setCookie('userPreference', 'darkMode', 60); // Expires in 60 minutes

getCookie(name)

Get a browser cookie value.

const preference = k.getCookie('userPreference');

getUrlParameter(name)

Get URL query parameter value.

// For URL: page.html?id=123
const id = k.getUrlParameter('id'); // Returns '123'

getPageRequest()

Get decoded page request object from URL.

// For URL: page.html?request=eyJtb2RlbCI6IlVzZXIiLCJpZHgiOjEyM30=
const request = k.getPageRequest();
// Returns: {model: "User", idx: 123}

setPageRequest(model, value)

Create encoded page request URL.

const url = k.setPageRequest('User', 123);
// Returns: ?request=eyJtb2RlbCI6IlVzZXIiLCJpZHgiOjEyM30=

alert(title, message, callback, dismiss)

Display a SweetAlert2 alert dialog.

k.alert('Success', 'Your changes have been saved!', () => {
    console.log('Alert dismissed');
});

confirm(title, message, callback, cancel)

Display a SweetAlert2 confirmation dialog.

k.confirm(
    'Delete User',
    'Are you sure you want to delete this user?',
    () => {
        console.log('User confirmed');
    },
    () => {
        console.log('User cancelled');
    }
);

validateForm(form)

Validate a form element (checks required fields).

const formElement = document.getElementById('myForm');
if (k.validateForm(formElement)) {
    console.log('Form is valid');
}

getNestedValue(obj, path)

Get nested object property using dot notation.

const user = { profile: { name: 'John' } };
const name = k.getNestedValue(user, 'profile.name'); // Returns 'John'
KyteTable Class - DataTables Integration
Constructor

Create a server-side DataTable with Kyte backend integration.

const userTable = new KyteTable(
    k,                      // Kyte instance
    '#usersTable',          // Table selector
    'User',                 // Model name
    [                       // Column definitions
        { data: 'id', title: 'ID' },
        { data: 'name', title: 'Name' },
        { data: 'email', title: 'Email' },
        { data: 'role_obj.name', title: 'Role' }
    ],
    true,                   // Sortable
    [[0, 'desc']],         // Default order
    true,                   // Show edit button
    true,                   // Show clickable rows
    'id',                   // ID field
    '/user/details.html'    // Detail page URL
);

Features:

  • Server-side pagination, sorting, and searching
  • Automatic foreign key resolution (e.g., role_obj.name)
  • Built-in edit/delete action buttons
  • Row click navigation
  • Auto-refresh on form submission
KyteForm Class - Dynamic Form Builder
Constructor

Create a dynamic form with validation and automatic CRUD operations.

const userForm = new KyteForm(
    k,                      // Kyte instance
    '#userForm',            // Form container selector
    'User',                 // Model name
    { kyte_account: 1 },   // Hidden fields
    [                       // Field definitions
        {
            field: 'name',
            type: 'text',
            label: 'Full Name',
            placeholder: 'Enter full name',
            required: true,
            col: 6
        },
        {
            field: 'email',
            type: 'text',
            label: 'Email Address',
            placeholder: 'user@example.com',
            required: true,
            col: 6
        },
        {
            field: 'role',
            type: 'select',
            label: 'Role',
            required: true,
            col: 6,
            option: {
                ajax: true,
                data_model_name: 'Role',
                data_model_field: 'name',
                data_model_value: 'id'
            }
        },
        {
            field: 'password',
            type: 'password',
            label: 'Password',
            required: true,
            col: 6
        }
    ],
    'User Form',            // Form title
    userTable,              // Optional table to refresh
    '#userModal',           // Optional modal selector
    '#openFormBtn'          // Optional trigger button
);

Field Types:

  • text - Text input
  • password - Password input
  • textarea - Multiline text
  • select - Dropdown (supports AJAX loading)
  • file - File upload (supports S3 direct upload)
  • date - Date picker
  • checkbox - Checkbox
  • radio - Radio buttons

Features:

  • Automatic validation
  • Bootstrap modal integration
  • Edit/create mode switching
  • AJAX-populated dropdowns
  • File upload with progress
  • Itemized/repeatable field groups
KyteNav Class - Navigation Bar
Constructor

Create a Bootstrap 5 navigation bar.

const nav = new KyteNav(
    '#navigation',          // Navigation container
    [                       // Menu items
        { label: 'Home', url: '/index.html' },
        { label: 'Dashboard', url: '/dashboard.html' },
        {
            label: 'Admin',
            children: [
                { label: 'Users', url: '/admin/users.html' },
                { label: 'Settings', url: '/admin/settings.html' }
            ]
        }
    ],
    () => {                 // Logout callback
        k.sessionDestroy(() => {
            window.location.href = '/login.html';
        });
    },
    'MyApp' // Brand HTML
);

    }

KyteCalendar Class - Date Range Selector
Constructor

Create a custom calendar with date range selection.

const calendar = new KyteCalendar(
    '#calendarContainer',   // Container selector
    4,                      // Number of rows (months)
    3,                      // Number of columns
    (startDate, endDate) => {
        console.log('Selected range:', startDate, 'to', endDate);
    }
);
KytePasswordRequirement Class - Password Validation
Constructor

Create real-time password validation with visual feedback.

const passwordChecker = new KytePasswordRequirement(
    '#passwordField',       // Password input selector
    {
        minLength: 8,
        requireUppercase: true,
        requireLowercase: true,
        requireNumber: true,
        requireSpecial: true
    }
);

Automatically displays validation checklist and updates in real-time as user types.

KyteWebComponent Class - Template Engine
Constructor

Simple mustache-style template engine for rendering dynamic content.

// HTML template
<div id="userCard">
    <h3>{{name}}</h3>
    <p>{{email}}</p>
    <p>Role: {{role_obj.name}}</p>
</div>

// JavaScript
const component = new KyteWebComponent(
    '#userCard',
    {
        name: 'John Doe',
        email: 'john@example.com',
        role_obj: { name: 'Admin' }
    },
    (data) => {
        // Optional mutator function
        data.name = data.name.toUpperCase();
        return data;
    }
);

Template Syntax:

  • {{'{{'}}property{{'}}'}} - Simple property
  • {{'{{'}}nested.property{{'}}'}} - Nested property
  • {{'{{'}}array[0]{{'}}'}} - Array access
Complete Example - User Management Page

Note: In published pages, k is automatically initialized. This example shows what your page JavaScript would look like:

// k is automatically created and initialized by Kyte:
// var k = new Kyte(endpoint, public_key, identifier, account_number, app_identifier);
// k.init();

// Check authentication
if (!k.isSession(false, false)) {
    window.location.href = '/login.html';
}

// Enable periodic session check
k.isSession(true, true, 30000);

// Create user table
const userTable = new KyteTable(
    k,                      // Use the global k instance
    '#usersTable',
    'User',
    [
        { data: 'id', title: 'ID' },
        { data: 'name', title: 'Name' },
        { data: 'email', title: 'Email' },
        { data: 'role_obj.name', title: 'Role' }
    ],
    true,
    [[0, 'desc']],
    true,
    true,
    'id',
    '/user/details.html'
);

// Create user form
const userForm = new KyteForm(
    k,
    '#userForm',
    'User',
    {},
    [
        {
            field: 'name',
            type: 'text',
            label: 'Full Name',
            required: true,
            col: 6
        },
        {
            field: 'email',
            type: 'text',
            label: 'Email',
            required: true,
            col: 6
        },
        {
            field: 'role',
            type: 'select',
            label: 'Role',
            required: true,
            col: 6,
            option: {
                ajax: true,
                data_model_name: 'Role',
                data_model_field: 'name',
                data_model_value: 'id'
            }
        },
        {
            field: 'password',
            type: 'password',
            label: 'Password',
            required: true,
            col: 6
        }
    ],
    'User Form',
    userTable,
    '#userModal',

// Create navigation

        { label: 'Dashboard', url: '/dashboard.html' },

        { label: 'Settings', url: '/settings.html' }

    () => {
        k.sessionDestroy(() => {
            window.location.href = '/login.html';
        });
    },
    '<strong>MyApp</strong>'
);
